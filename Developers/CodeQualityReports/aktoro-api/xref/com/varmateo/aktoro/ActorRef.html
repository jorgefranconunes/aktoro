10  /**
11   * A wrapper for an actor instance.
12   *
13   * <p>Instances of this class are tipically used only inside the class
14   * implementing an actor. The <code>ActorRef</code> allows the actor
15   * to execute lambdas and callbacks without closing on its internal
16   * context.</p>
17   *
18   * <p>When an actor is created an <code>ActorRef</code> instance is
19   * passed to its factory (see
20   * <code>{@link ActorSystem#createActor(Class,ActorFactory)}</code>. The
21   * actor can then use that <code>ActorRef</code> to execute lambdas,
22   * by calling <code>{@link #perform(Runnable)}</code>, or to pass references
23   * of itself to outside, by calling <code>{@link #self()}</code>.</p>
24   *
25   * @param T The type of the actor.
26   */
27  public interface ActorRef<T> {
28  
29  
30      /**
31       * Performs an action in the scope of the associated actor. The
32       * given action will be executed as if it had been called as a
33       * method of this actor. This means that when the given action is
34       * being executed no other thread will be executing methods of the
35       * actor. Also, this method dos not block, and returns without
36       * waiting for the action to complete.
37       *
38       * @param action The action to be executed.
39       */
40      void perform(Runnable action);
41  
42  
43      /**
44       * Fetches the actor for this <code>ActorRef</code>.
45       *
46       * @return The associated actor.
47       */
48      T self();
49  
50  
51  }
