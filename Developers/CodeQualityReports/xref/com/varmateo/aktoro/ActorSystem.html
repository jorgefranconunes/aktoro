7   package com.varmateo.aktoro;

9   import com.varmateo.aktoro.ActorCoreFactory;
10  
11  
12  /**
13   * Contract for a manager of a set of actors.
14   */
15  public interface ActorSystem {
16  
17  
18      /**
19       * Creates a new actor.
20       *
21       * <p>A new actor instance is created through the provided
22       * <code>{@link ActorCoreFactory}</code>. A proxy for that actor
23       * instance will then be returned. The actor proxy is the visible
24       * face of the actor for the outside world. The actor proxy,
25       * working within the actor system, has the responsability to
26       * ensure the actor methods are called in sequence, and never
27       * concurrently.</p>
28       *
29       * @param <T> Type of objects to be created. This must be an
30       * interface type.
31       *
32       * @param actorCoreFactory Used for creating an actual actor core
33       * instance.
34       *
35       * @param actorType The class object of the interface the actor
36       * instance implements. This will also be the type of the returned
37       * proxy actor.
38       *
39       * @return An object implementing the given <code>actorType</code>
40       * interface. That object is a proxy for the actor instance
41       * created with the <code>ActorCoreFactory</code>.
42       */
43      <T> T createActor(
44              ActorCoreFactory<T> actorCoreFactory,
45              Class<T> actorType);
46  
47  }
